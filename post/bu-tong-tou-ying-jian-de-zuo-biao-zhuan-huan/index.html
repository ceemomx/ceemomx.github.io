不同投影间的坐标转换

2022-04-25 1 min read

使用proj4.js进行转换

投影可以去epsg.io进行查询

const proj4 = require('proj4')

//定义源坐标的投影，如果源目标投影是非常用投影就需要定义，使用4528示例
proj4.defs('EPSG:4528', '+proj=tmerc +lat_0=0 +lon_0=120 +k=1 +x_0=40500000 +y_0=0 +ellps=GRS80 +units=m +no_defs')

//定义目标坐标投影，使用4490示例
proj4.defs('EPSG:4490', '+proj=longlat +ellps=GRS80 +no_defs')

let x = 40529671.21
let y = 4010004.59

let coor = proj4('EPSG:4528', 'EPSG:4490', [x, y])

console.log(coor)
// [120.33, 36.22]
